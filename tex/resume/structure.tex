%----------------------------------------------------------------------------------------
%	MAIN HEADER COMMAND
%----------------------------------------------------------------------------------------

\newcommand{\customheader}[6]{
    \parbox{0.6\textwidth}
    {
        {\huge{\color{slate-blue}\textbf{#1}}}\\ % Header  name and color
        {\Large\color{light-black}{#2}}\\
    }
    \hfill % Horizontal space between the two blocks
    \parbox{0.4\textwidth}
    {
        \begin{tabbing} % Enables tabbing
            \color{light-black}{#3} \\
            \color{light-black}{#4} \\
            \color{light-black}{#5} \\
            \color{light-black}{#6} \\
        \end{tabbing}
    }
}

%----------------------------------------------------------------------------------------
%	SUMMARY SECTION
%----------------------------------------------------------------------------------------

\newcommand{\summary}[2]{
    \vspace{-2em}\section{#1}
    \color{light-black}
    {#2}
}

%----------------------------------------------------------------------------------------
%	SKILL GROUP COMMAND
%----------------------------------------------------------------------------------------

\newcommand{\skillgroup}[3]{
    \begin{tabbing}
    \hspace{0.5cm}\color{dark-black}{#1}\\
    \ifx\temp\empty
    \hspace{1cm}{#2}
    \else
    \hspace{1cm}{#2}\\
    \hspace{1cm}{#3}
    \fi
    \end{tabbing}
    \vspace{-7mm}
}

%----------------------------------------------------------------------------------------
%	DATE ITEM COMMAND
%----------------------------------------------------------------------------------------

\newcommand{\dateitem}[3]{

}
